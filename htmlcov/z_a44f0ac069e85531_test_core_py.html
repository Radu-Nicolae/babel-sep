312@pytest.mark.parametrize('filename', [ 

313 'babel/global.dat',

322def test_compatible_classes_in_global_and_localedata(filename): 

323 import pickle 

324 

325 class Unpickler(pickle.Unpickler): 

326 

327 def find_class(self, module, name): 

328 # *.dat files must have compatible classes between Python 2 and 3 

329 if module.split('.')[0] == 'babel': 

330 return pickle.Unpickler.find_class(self, module, name) 

331 raise pickle.UnpicklingError(f"global '{module}.{name}' is forbidden") 

332 

333 with open(filename, 'rb') as f: 

334 assert Unpickler(f).load()
